\title{Two orthogonal factorization systems on ${\bf COB}$}

\author{David I. Spivak}

\maketitle

\begin{definition}

Let $W$ be an oriented 1-cobordism. Then $W$ can be uniquely written as the disjoint union of an {\em interval part} $W^|$and a {\em closed loop part} $W^\circ$, where every component of $W^|$ is diffeomorphic to the oriented unit interval and every component of $W^\circ$ is diffeomorphic to the oriented unit circle.

\end{definition}

\begin{lemma}

Let $X$ and $Y$ be signed sets, and let $W$ be a 1-cobordism from $X$ to $Y$. The closed loop part of $W$ can be identified with a natural number $\mcL(W)\in\NN$. The interval part of $W$ can be identified with a bijection of the form
$$\inp{X}+\outp{Y}\To{\iso}\outp{X}+\inp{Y}.$$

\end{lemma}

\begin{definition}

Let $X=(\inp{X},\outp{X})$ and $Y=(\inp{Y},\outp{Y})$ be signed sets, and let $f\taking\inp{X}+\outp{Y}\To{\iso}\outp{X}+\inp{Y}$ be the  bijection signfying the interval part of a cobordism $\Phi$. Then $\Phi$ is called a {\em cofibration} if $f$ is the sum of two functions of the form
$$
\begin{cases}
\outp{f}\taking\outp{Y}\to\outp{X}\\
\inp{f}\taking\inp{X}\to\outp{X}+\inp{Y}
\end{cases}
$$
and $\Phi$ is called a {\em fibration} if $f$ is the sum of two functions of the form
$$
\begin{cases}
\outp{f}\taking\outp{Y}\to\outp{X}+\inp{Y}\\
\inp{f}\taking\inp{X}\to\inp{Y}
\end{cases}
$$

\end{definition}

\begin{proposition}
Let $W=(\inp{W},\outp{W}), X=(\inp{X},\outp{X}), Y=(\inp{Y},\outp{Y}),$ and $Z=(\inp{Z},\outp{Z}),$ be signed sets, and let 
$$W\To{e}X\To{\ell}Y\To{m}Z$$
be composable morphisms in $\Cob$. If $e$ is a cofibration and $m$ is a fibration, then the interval part of the composition can be computed as
$$\inp{W}+\outp{Z}\To{\inp{e}+\outp{m}}\outp{W}+\inp{X}+\outp{Y}+\inp{Z}\To{\;\ell\;}\outp{W}+\outp{X}+\inp{Y}+\inp{Z}\To{\outp{e}+\inp{m}}\outp{W}+\inp{Z}.$$
The closed loop part of such a composition is simply given by the usual sum 
$$\mcL(m\circ\ell\circ e):=\mcL(e)+\mcL(\ell)+\mcL(m)\in\NN.$$

\end{proposition}
